    float price
    int   bar
    float vol
    float wick

type Zone
    float top
    float bot
    float center
    float strength
    float volSum

plPx = ta.pivotlow(low,   pivLeft, pivRight)

upperWick(int idx) =>
    high[idx] - math.max(open[idx], close[idx])

lowerWick(int idx) =>
    math.min(open[idx], close[idx]) - low[idx]

safeVolume(int idx) =>
    v = volume[idx]
    na(v) or v <= 0 ? 1.0 : v

var array<Pivot> lowPivots  = array.new<Pivot>()

if not na(plPx) and plPx > 0 and dataReady
    lowPivots.push(Pivot.new(plPx, bar_index - pivRight, safeVolume(pivRight), lowerWick(pivRight)))
    if lowPivots.size() > pivBuffer
        lowPivots.shift()
    newLowPivot := true

// =====================================================================
//                            CLUSTERING
// =====================================================================

// Composite pivot weight:
//   weight = (1 + 0.5 * volume_ratio + 0.5 * wick_ratio) * decay(age)
//   - volume_ratio  = pivot volume / SMA(volume, 50)        : heavier candles count more
//   - wick_ratio    = rejection wick length / ATR(50)       : longer rejection wicks count more
//   - decay         = 1 / (1 + age_in_bars / 200)           : older pivots gradually fade
// The weighted sum of pivots in a cluster gives both the zone's center price
// (weighted average) and its relative strength score used for ranking.
pivotWeight(Pivot p, float a, float vAvg) =>
    safeVol  = na(p.vol)  or p.vol  <= 0 ? 0.0 : p.vol
    safeWick = na(p.wick) or p.wick <  0 ? 0.0 : p.wick
    age      = math.max(1, bar_index - p.bar)
    decay    = 1.0 / (1.0 + age / 200.0)
    volNorm  = vAvg > 0 ? safeVol / vAvg : 1.0
    wickNorm = a > 0 ? safeWick / a : 0.0
    (1.0 + volNorm * 0.5 + wickNorm * 0.5) * decay

buildZones(array<Pivot> src, bool resistance, float a, float vAvg) =>
    out = array.new<Zone>()
    if src.size() >= minTouches and a > 0
        used = array.new<bool>()
        for i = 0 to src.size() - 1
            used.push(false)
        tol = a * clusterTol
        for i = 0 to src.size() - 1
            if not used.get(i)
                seed    = src.get(i)
                members = array.new<Pivot>()
                members.push(seed)
                used.set(i, true)
                if i + 1 <= src.size() - 1
                    for j = i + 1 to src.size() - 1
                        if not used.get(j)
                            cand = src.get(j)
                            if math.abs(cand.price - seed.price) <= tol
                                farEnough = true
                                if minSpacing > 0
                                    for k = 0 to members.size() - 1
                                        if math.abs(cand.bar - members.get(k).bar) < minSpacing
                                            farEnough := false
                                            break
                                if farEnough
                                    members.push(cand)
                                    used.set(j, true)
                if members.size() >= minTouches
                    priceSum  = 0.0
                    weightSum = 0.0
                    volSum    = 0.0
                    earliest  = members.get(0).bar
                    latest    = members.get(0).bar
                    for k = 0 to members.size() - 1
                        m = members.get(k)
                        w = pivotWeight(m, a, vAvg)
                        priceSum  += m.price * w
                        weightSum += w
                        volSum    += na(m.vol) ? 0.0 : m.vol
                        earliest  := math.min(earliest, m.bar)
                        latest    := math.max(latest,   m.bar)
                    if weightSum > 0
                        center = priceSum / weightSum
                        pad    = a * 0.25
                        out.push(Zone.new(center + pad, center - pad, center, weightSum, volSum, members.size(), earliest, latest, na, resistance, false, false))
    out

trimToTopN(array<Zone> zones, int keep) =>
    result = array.new<Zone>()
    if zones.size() > 0
        copy = zones.copy()
        limit = math.min(keep, copy.size())
        for i = 0 to limit - 1
            if copy.size() == 0
                break
            bestIdx = 0
            bestVal = copy.get(0).strength
            if copy.size() > 1
                for j = 1 to copy.size() - 1
                    if copy.get(j).strength > bestVal
                        bestVal := copy.get(j).strength
                        bestIdx := j
            result.push(copy.get(bestIdx))
            copy.remove(bestIdx)
    result

// =====================================================================
//                       ZONE STATE & ARCHIVING
// =====================================================================
var array<Zone> resZones    = array.new<Zone>()
var array<Zone> supZones    = array.new<Zone>()
var array<Zone> brokenZones = array.new<Zone>()

archiveIfNew(Zone z, float a) =>
    duplicate = false
    if brokenZones.size() > 0
        tol = a * clusterTol * 0.7
        for i = 0 to brokenZones.size() - 1
            existing = brokenZones.get(i)
            if existing.isResistance == z.isResistance and math.abs(existing.center - z.center) <= tol
                duplicate := true
                if z.wasBroken and not existing.wasBroken
                    existing.wasBroken   := true
                    existing.diedBar     := z.diedBar
                    existing.isLiveBreak := z.isLiveBreak
                    brokenZones.set(i, existing)
                break
    if not duplicate
        brokenZones.push(z)
        if brokenZones.size() > 100
            brokenZones.shift()
    duplicate

// Live-break check: a break only counts if price was on the correct side
// of the zone on the previous close. This filters out retroactive zone
// formations during sharp moves (the "born-dead" case).
if resZones.size() > 0
    for i = 0 to resZones.size() - 1
        z = resZones.get(i)
        if na(z.diedBar) and close > z.top
            z.diedBar     := bar_index
            z.wasBroken   := true
            z.isLiveBreak := close[1] <= z.top
            resZones.set(i, z)
            archiveIfNew(z, atr)

if supZones.size() > 0
    for i = 0 to supZones.size() - 1
        z = supZones.get(i)
        if na(z.diedBar) and close < z.bot
            z.diedBar     := bar_index
            z.wasBroken   := true
            z.isLiveBreak := close[1] >= z.bot
            supZones.set(i, z)
            archiveIfNew(z, atr)

if resZones.size() > 0
    for i = 0 to resZones.size() - 1
        z = resZones.get(i)
        if na(z.diedBar) and high >= z.bot and low <= z.top
            z.lastTouch := bar_index
            resZones.set(i, z)

if supZones.size() > 0
    for i = 0 to supZones.size() - 1
        z = supZones.get(i)
        if na(z.diedBar) and high >= z.bot and low <= z.top
            z.lastTouch := bar_index
            supZones.set(i, z)

zoneSurvives(Zone old, array<Zone> newSet, float a) =>
    survived = false
    if newSet.size() > 0
        tol = a * clusterTol
        for i = 0 to newSet.size() - 1
            nz = newSet.get(i)
            if math.abs(nz.center - old.center) <= tol
                survived := true
                break
    survived

isAlreadyBroken(Zone z, float a) =>
    broken = false
    if brokenZones.size() > 0
        tol = a * clusterTol * 0.7
        for i = 0 to brokenZones.size() - 1
            bz = brokenZones.get(i)
            if bz.isResistance == z.isResistance and bz.wasBroken and math.abs(bz.center - z.center) <= tol
                broken := true
                break
    broken

inheritState(Zone newZ, array<Zone> oldZones, float a) =>
    inherited = newZ
    if oldZones.size() > 0
        tol = a * clusterTol
        for i = 0 to oldZones.size() - 1
            oldZ = oldZones.get(i)
            if math.abs(oldZ.center - newZ.center) <= tol
                inherited.lastTouch := oldZ.lastTouch
                break
    inherited

if newHighPivot or newLowPivot
    newRes = trimToTopN(buildZones(highPivots, true,  atr, volMa), zoneCount)
    newSup = trimToTopN(buildZones(lowPivots,  false, atr, volMa), zoneCount)
    if resZones.size() > 0
        for i = 0 to resZones.size() - 1
            old = resZones.get(i)
            if na(old.diedBar) and old.touches >= minTouches and not zoneSurvives(old, newRes, atr)
                old.diedBar   := bar_index
                old.wasBroken := false
                archiveIfNew(old, atr)
    if supZones.size() > 0
        for i = 0 to supZones.size() - 1
            old = supZones.get(i)
            if na(old.diedBar) and old.touches >= minTouches and not zoneSurvives(old, newSup, atr)
                old.diedBar   := bar_index
                old.wasBroken := false
                archiveIfNew(old, atr)
    oldResCopy = resZones.copy()
    oldSupCopy = supZones.copy()
    resZones.clear()
    if newRes.size() > 0
        for i = 0 to newRes.size() - 1
            nz = newRes.get(i)
            if not isAlreadyBroken(nz, atr)
                inh = inheritState(nz, oldResCopy, atr)
                resZones.push(inh)
    supZones.clear()
    if newSup.size() > 0
        for i = 0 to newSup.size() - 1
            nz = newSup.get(i)
            if not isAlreadyBroken(nz, atr)
                inh = inheritState(nz, oldSupCopy, atr)
                supZones.push(inh)

var array<float> sigZoneCenters = array.new<float>()
var array<int>   sigZoneBars    = array.new<int>()

cooldownOk(float center) =>
    ok = true
    if sigZoneCenters.size() > 0
        for i = 0 to sigZoneCenters.size() - 1
            if math.abs(sigZoneCenters.get(i) - center) <= atr * clusterTol
                if bar_index - sigZoneBars.get(i) < sigCooldown
                    ok := false
                    break
    ok

recordSignal(float center) =>
    sigZoneCenters.push(center)
    sigZoneBars.push(bar_index)
    if sigZoneCenters.size() > 50
        sigZoneCenters.shift()
        sigZoneBars.shift()
    center

zoneRenderable(Zone z) =>
    na(z.diedBar) and z.touches >= minTouches and not na(z.top) and not na(z.bot) and z.top > 0 and z.bot > 0 and z.bot < z.top

checkSupBounce() =>
    fired = false
    if supZones.size() > 0
        for i = 0 to supZones.size() - 1
            z = supZones.get(i)
            if zoneRenderable(z)
                pierced = low <= z.top
                heldUp  = close >= z.bot and close >= open
                if pierced and heldUp and cooldownOk(z.center)
                    recordSignal(z.center)
                    fired := true
                    break
    fired

checkResReject() =>
    fired = false
    if resZones.size() > 0
        for i = 0 to resZones.size() - 1
            z = resZones.get(i)
            if zoneRenderable(z)
                pierced = high >= z.bot
                heldDn  = close <= z.top and close <= open
                if pierced and heldDn and cooldownOk(z.center)
                    recordSignal(z.center)
                    fired := true
                    break
    fired

checkRetestBull() =>
    fired = false
    if brokenZones.size() > 0
        for i = 0 to brokenZones.size() - 1
            z = brokenZones.get(i)
            if z.isResistance and z.wasBroken and not na(z.diedBar)
                age = bar_index - z.diedBar
                if age > 0 and age <= retestWindow
                    if low <= z.top and close >= z.bot and close >= open
                        fired := true
                        break
    fired

checkRetestBear() =>
    fired = false
    if brokenZones.size() > 0
        for i = 0 to brokenZones.size() - 1
            z = brokenZones.get(i)
            if not z.isResistance and z.wasBroken and not na(z.diedBar)
                age = bar_index - z.diedBar
                if age > 0 and age <= retestWindow
                    if high >= z.bot and close <= z.top and close <= open
                        fired := true
                        break
    fired

bounceUp   = checkSupBounce()
bounceDown = checkResReject()
retestUp   = checkRetestBull()
retestDown = checkRetestBear()

plotshape(showTest   and bounceUp,   "Successful test up",   shape.circle,  location.belowbar, color.new(supColor, 0), size = size.tiny)
plotshape(showTest   and bounceDown, "Successful test down", shape.circle,  location.abovebar, color.new(resColor, 0), size = size.tiny)
plotshape(showRetest and retestUp,   "Retest support",       shape.diamond, location.belowbar, color.new(supColor, 0), size = size.tiny)
plotshape(showRetest and retestDown, "Retest resistance",    shape.diamond, location.abovebar, color.new(resColor, 0), size = size.tiny)

resHold = false
supHold = false
if resZones.size() > 0
    for i = 0 to resZones.size() - 1
        z = resZones.get(i)
        if zoneRenderable(z)
            if high >= z.bot and close < z.top
                resHold := true
                break
if supZones.size() > 0
    for i = 0 to supZones.size() - 1
        z = supZones.get(i)
        if zoneRenderable(z)
            if low <= z.top and close > z.bot
                supHold := true
                break

plotshape(showReaction and supHold, "Support hold",    shape.diamond, location.belowbar, color.new(supColor, 30), size = size.tiny)
plotshape(showReaction and resHold, "Resistance hold", shape.diamond, location.abovebar, color.new(resColor, 30), size = size.tiny)

var array<box>   bxBuf = array.new<box>()
var array<line>  lnBuf = array.new<line>()
var array<label> lbBuf = array.new<label>()

clearBuffers() =>
    if bxBuf.size() > 0
        for i = 0 to bxBuf.size() - 1
            box.delete(bxBuf.get(i))
    if lnBuf.size() > 0
        for i = 0 to lnBuf.size() - 1
            line.delete(lnBuf.get(i))
    if lbBuf.size() > 0
        for i = 0 to lbBuf.size() - 1
            label.delete(lbBuf.get(i))
    bxBuf.clear()
    lnBuf.clear()
    lbBuf.clear()
    bxBuf.size()

strengthStars(float s, float maxS) =>
    ratio = maxS > 0 ? s / maxS : 0.0
    ratio >= 0.85 ? "★★★" : ratio >= 0.55 ? "★★" : "★"

ageFade(int lastTouch) =>
    if fadeAge <= 0
        0
    else
        age = bar_index - lastTouch
        age <= 0 ? 0 : math.min(15, int(age * 15.0 / fadeAge))

isFar(float zoneCenter) =>
    if hideFar <= 0 or close <= 0
        false
    else
        distPct = math.abs(zoneCenter - close) / close * 100.0
        distPct > hideFar

// Clamp a historical bar index to a safe offset within Pine's buffer.
// Pine's max_bars_back limit is 5000; we leave a small safety margin.
safeStartBar(int b) =>
    minSafe = bar_index - 4900
    b < minSafe ? minSafe : b

formatVolume(float v) =>
    if v >= 1000000000
        str.tostring(v / 1000000000, "#.##") + "B"
    else if v >= 1000000
        str.tostring(v / 1000000, "#.##") + "M"
    else if v >= 1000
        str.tostring(v / 1000, "#.##") + "K"
    else
        str.tostring(v, "#.##")

drawAlive(Zone z, color baseCol, float maxS, float maxVol) =>
    if not na(z.top) and not na(z.bot) and not na(z.center) and z.top > 0 and z.bot > 0 and z.bot < z.top and not isFar(z.center)
        rightX  = bar_index + 25
        startX  = safeStartBar(z.firstBar)
        extra   = ageFade(z.lastTouch)
        volBoost = 0
        if useVolColor and maxVol > 0
            volRatio = math.min(1.0, z.volSum / maxVol)
            volBoost := int(volRatio * 35)
        fillOp = math.max(40, math.min(95, zoneOpacity + extra - volBoost))
        fill   = color.new(baseCol, fillOp)
        edge   = color.new(baseCol, math.min(95, 35 + extra - volBoost / 2))
        bxBuf.push(box.new(startX, z.top, rightX, z.bot, border_color = na, bgcolor = fill))
        lnBuf.push(line.new(startX, z.center, rightX, z.center, color = edge, style = line.style_dashed, width = 1))
        if showLabels
            string txt = ""
            stars = strengthStars(z.strength, maxS)
            if showVolText and maxVol > 0
                string volVal = ""
                if volFormat == "Percentage"
                    volPct = int((z.volSum / maxVol) * 100)
                    volVal := str.tostring(volPct) + "%"
                else
                    volVal := formatVolume(z.volSum)
                if singleLine
                    txt := stars + "  " + str.tostring(z.touches) + "×  •  Vol: " + volVal
                else
                    txt := "Vol: " + volVal + "\n" + stars + "  " + str.tostring(z.touches) + "×"
            else
                txt := stars + "  " + str.tostring(z.touches) + "×"
            lbBuf.push(label.new(rightX, z.center, txt, style = label.style_label_left, color = color.new(baseCol, 75), textcolor = chart.fg_color, size = size.small))
    z.center

drawBroken(Zone z) =>
    if not na(z.top) and not na(z.bot) and z.top > 0 and z.bot > 0 and z.bot < z.top and not isFar(z.center)
        useCol  = z.isResistance ? brokenBear : brokenBull
        fill    = color.new(useCol, brokenOp)
        edge    = color.new(useCol, math.min(90, brokenOp + 10))
        startX  = safeStartBar(z.firstBar)
        endBar  = na(z.diedBar) ? bar_index : safeStartBar(z.diedBar)
        bxBuf.push(box.new(startX, z.top, endBar, z.bot, border_color = na, bgcolor = fill))
        lnBuf.push(line.new(startX, z.center, endBar, z.center, color = edge, style = line.style_dashed, width = 1))
        if showReaction and z.wasBroken and not na(z.diedBar) and z.isLiveBreak
            lblY     = z.isResistance ? z.top : z.bot
            lblStyle = z.isResistance ? label.style_label_down : label.style_label_up
            lblTxt   = z.isResistance ? "Break R" : "Break S"
            lblColor = z.isResistance ? color.rgb(200, 60, 60) : color.rgb(40, 160, 120)
            lbBuf.push(label.new(safeStartBar(z.diedBar), lblY, lblTxt, style = lblStyle, color = lblColor, textcolor = color.white, size = size.small))
    z.center

maxStrengthAcross() =>
    m = 0.0
    if resZones.size() > 0
        for i = 0 to resZones.size() - 1
            z = resZones.get(i)
            if na(z.diedBar)
                m := math.max(m, z.strength)
    if supZones.size() > 0
        for i = 0 to supZones.size() - 1
            z = supZones.get(i)
            if na(z.diedBar)
                m := math.max(m, z.strength)
    m

maxVolAcross() =>
    m = 0.0
    if resZones.size() > 0
        for i = 0 to resZones.size() - 1
            z = resZones.get(i)
            if na(z.diedBar)
                m := math.max(m, z.volSum)
    if supZones.size() > 0
        for i = 0 to supZones.size() - 1
            z = supZones.get(i)
            if na(z.diedBar)
                m := math.max(m, z.volSum)
    m

if barstate.islast
    clearBuffers()

    mv = maxVolAcross()
    if resZones.size() > 0

    if (showBroken or showDropped) and brokenZones.size() > 0
        for i = 0 to brokenZones.size() - 1

            shouldShow = (z.wasBroken and showBroken) or (not z.wasBroken and showDropped)

priceTouchesAlive(array<Zone> zs) =>
    hit = false
    if zs.size() > 0
        for i = 0 to zs.size() - 1
            z = zs.get(i)
            if na(z.diedBar) and z.touches >= minTouches and high >= z.bot and low <= z.top
                hit := true
                break
    hit

touchedRes = priceTouchesAlive(resZones)
touchedSup = priceTouchesAlive(supZones)

brokeRes = false
brokeSup = false
if resZones.size() > 0
    for i = 0 to resZones.size() - 1
        z = resZones.get(i)
        if not na(z.diedBar) and z.diedBar == bar_index and z.touches >= minTouches and z.isLiveBreak
            brokeRes := true
if supZones.size() > 0
    for i = 0 to supZones.size() - 1
        z = supZones.get(i)
        if not na(z.diedBar) and z.diedBar == bar_index and z.touches >= minTouches and z.isLiveBreak
            brokeSup := true
